    import torch
import matplotlib.pyplot as plt
import numpy as np
from d2lzh_pytorch import *

lr=0.03
num_epoches=10
batch_size=10
num_inputs=2
num_examples=1000
true_w=[1.2,3]
true_b=2.3
features=torch.from_numpy(np.random.normal(0,1,(num_examples,num_inputs)))
labels=features[:,0]*true_w[0]+features[:,1]*true_w[1]+true_b
labels+=torch.from_numpy(np.random.normal(0,0.01,labels.size()))

set_figsize()
#plt.scatter(features[:,0].numpy(),labels.numpy())
#plt.show()
w=torch.tensor(np.random.normal(0,0.01,(num_inputs,1)),dtype=torch.float64)
b=torch.zeros(1,dtype=torch.float64)
w.requires_grad=True
b.requires_grad=True
def linml(X,w,b):
    return torch.mm(X,w)+b
def MSEloss(y,y_hat):
    return (y_hat-y.view(y_hat.size()))**2
def optimizer(params,lr,batch_size):
    for param in params:
        param.data-=lr*param.grad/batch_size
for epoch in range(num_epoches):
    for X,y in data_iter(batch_size,features,labels):
        l=MSEloss(y,linml(X,w,b)).sum()
        l.backward()
        optimizer([w,b],lr,batch_size)
        w.grad.data.zero_()
        b.grad.data.zero_()
    print(f"第{epoch}轮的偏差为{l}")
print(f"true_w:{true_w},true_b:{true_b}\n w:{w.data},b:{b.data}")

    简约实现：¶

    import torch
import torch.utils.data as Data
import torch.nn as nn
from torch.nn import init
import torch.optim as optim
import matplotlib.pyplot as plt
import numpy as np
from d2lzh_pytorch import *

lr=0.03
num_epoches=10
batch_size=10
num_inputs=2
num_examples=1000
true_w=[1.2,3]
true_b=2.3
features=torch.from_numpy(np.random.normal(0,1,(num_examples,num_inputs)))
labels=features[:,0]*true_w[0]+features[:,1]*true_w[1]+true_b
labels+=torch.from_numpy(np.random.normal(0,0.01,labels.size()))

data_set=Data.TensorDataset(features,labels)
data_iter=Data.DataLoader(data_set,batch_size,shuffle=True)

class Linear_net(nn.Module):
    def __init__(self,n_features):
        super(Linear_net,self).__init__()
        self.linear=nn.Linear(n_features,1)
    def forward(self,x):
        return self.linear(x.float())

net=Linear_net(num_inputs)
print(net)
init.normal(net.linear.weight,mean=0,std=0.01)
init.constant(net.linear.bias,val=0)

loss=nn.MSELoss()

optimizer=optim.SGD(net.parameters(),lr=0.03)

for epoch in range(1,num_epoches+1):
    for X,y in data_iter:
        output=net(X)
        l=loss(output,y.view(-1,1).float())
        optimizer.zero_grad()
        l.backward()
        optimizer.step()
    print(f"第{epoch}轮,损失为{l.item()}")

dense=net.linear
print(dense.weight,dense.bias)
